{{/* Renders description HTML: if source contains backticks, first escape only outside backticks, then markdownify + escape inside . Otherwise, escape in source, then markdownify. Use printf "<%s>" "$1" so Hugo passes literal $1 to the regex (not template var $1). */}} {{- $desc := default "" . }} {{- $repl := printf "<%s>" "$1" }} {{- if in $desc "`" }} {{- $backtickBlocks := findRE "`[^`]+`" $desc }} {{- $work := $desc }} {{- range $idx, $block := $backtickBlocks }} {{- $work = replace $work $block (printf "{{BT%d}}" $idx) }} {{- end }} {{- $work = replaceRE `<([a-zA-Z][a-zA-Z0-9-]*)>` $repl $work }} {{- range $idx, $block := $backtickBlocks }} {{- $work = replace $work (printf "{{BT%d}}" $idx) $block }} {{- end }} {{- partial "openapi/escape-code-blocks" ($work | markdownify) }} {{- else }} {{- $withEscapedPlaceholders := replaceRE `<([a-zA-Z][a-zA-Z0-9-]*)>` $repl $desc }} {{- $withEscapedPlaceholders | markdownify }} {{- end }}
