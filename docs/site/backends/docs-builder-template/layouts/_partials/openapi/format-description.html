{{/* Renders description HTML: if source contains backticks, first escape only outside backticks, then markdownify + escape inside . Otherwise, escape in source, then markdownify. Use printf "<%s>" "$1" so Hugo passes literal $1 to the regex (not template var $1). Placeholder for backtick blocks uses U+E000 (PUA) so literal "{{BT0}}" etc. in the description is not corrupted. */}} {{- $desc := default "" . }} {{- $repl := printf "<%s>" "$1" }} {{- $replClosing := printf "</%s>" "$1" }} {{- $sentinel := printf "%c" 57344 }} {{- if in $desc "`" }} {{- /* Use [^`]* (not +) so empty backtick pairs `` are matched and protected; else e.g. "`` `foo`" would match "` `" first and the second block would get escaped */}} {{- $backtickBlocks := findRE "`[^`]*`" $desc }} {{- $work := $desc }} {{- range $idx, $block := $backtickBlocks }} {{- $work = replace $work $block (printf "%sOPENAPI_BT_%d%s" $sentinel $idx $sentinel) }} {{- end }} {{- $work = replaceRE `<([a-zA-Z][a-zA-Z0-9-_]*)>` $repl $work }} {{- $work = replaceRE `` $replClosing $work }} {{- range $idx, $block := $backtickBlocks }} {{- $work = replace $work (printf "%sOPENAPI_BT_%d%s" $sentinel $idx $sentinel) $block }} {{- end }} {{- partial "openapi/escape-code-blocks" ($work | markdownify) }} {{- else }} {{- $withEscapedPlaceholders := replaceRE `<([a-zA-Z][a-zA-Z0-9-]*)>` $repl $desc }} {{- $withEscapedPlaceholders = replaceRE `` $replClosing $withEscapedPlaceholders }} {{- $withEscapedPlaceholders | markdownify }} {{- end }}
