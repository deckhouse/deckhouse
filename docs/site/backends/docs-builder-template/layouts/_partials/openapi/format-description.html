Otherwise, in the source replace only placeholder-like with <word> (so "> **bold**" and other markdown stay intact), then markdownify. */}} {{- $desc := default "" . }} {{- if in $desc "`" }} {{- partial "openapi/escape-code-blocks" ($desc | markdownify) }} {{- else }} {{- $withEscapedPlaceholders := replaceRE `<([a-zA-Z][a-zA-Z0-9-]*)>` "<$1>" $desc }} {{- $withEscapedPlaceholders | markdownify }} {{- end }}
