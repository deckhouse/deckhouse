{{/* Escapes < and > to < and > inside

...

blocks only (content between the tags), so that plain text like 'secret//' in descriptions displays correctly. Pass the output of markdownify (HTML string). */}} {{- $html := default "" . }} {{- $pBlocks := findRE `(?s)

.*?

` $html }} {{- range $pBlocks }} {{- $inner := replaceRE `(?s)

(.*?)

` "$1" . }} {{- $escaped := $inner | replaceRE "<" "<" | replaceRE ">" ">" }} {{- $fixed := printf "

%s

" $escaped }} {{- $html = replace $html . $fixed }} {{- end }} {{- $html | safeHTML }}
