{{- /* Formats a version requirement string. If the version starts with ">=", formats it as "version and above". If the version starts with ">", formats it as "above version". Otherwise, returns the version as-is. Usage: {{ partial "format-version-requirement" ">= 1.27" }} {{ partial "format-version-requirement" "> 1.27" }} */ -}} {{- $version := . | strings.TrimSpace }} {{- if strings.HasPrefix $version ">=" }} {{- $versionNum := strings.TrimPrefix ">=" $version | strings.TrimSpace }} {{- printf "%s %s" $versionNum (T "requirements_version_and_above") }} {{- else if strings.HasPrefix $version ">" }} {{- $versionNum := strings.TrimPrefix ">" $version | strings.TrimSpace }} {{- printf "%s %s" (T "requirements_version_above") $versionNum }} {{- else }} {{- $version }} {{- end -}}
